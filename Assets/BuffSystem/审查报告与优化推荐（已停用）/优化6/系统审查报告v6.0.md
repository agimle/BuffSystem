#### 5.2.2 分层更新频率优化（高优先级）

```csharp
public enum UpdateFrequency
{
    EveryFrame = 0,     // 视觉相关 (60fps)
    Every30ms = 1,      // 高频逻辑 (30fps)
    Every100ms = 2,     // 中频逻辑 (10fps)
    Every500ms = 3,     // 低频逻辑 (2fps)
    OnEventOnly = 4     // 事件驱动
}

public class BuffContainer
{
    // 按更新频率分桶
    private List<BuffEntity>[] updateBuckets;
    
    public void Update(float deltaTime)
    {
        // 每帧更新
        UpdateBucket(UpdateFrequency.EveryFrame, deltaTime);
        
        // 30ms更新
        if (timeAccumulator30ms >= 0.03f)
        {
            UpdateBucket(UpdateFrequency.Every30ms, timeAccumulator30ms);
            timeAccumulator30ms = 0;
        }
        
        // 100ms更新
        if (timeAccumulator100ms >= 0.1f)
        {
            UpdateBucket(UpdateFrequency.Every100ms, timeAccumulator100ms);
            timeAccumulator100ms = 0;
        }
        
        // 500ms更新
        if (timeAccumulator500ms >= 0.5f)
        {
            UpdateBucket(UpdateFrequency.Every500ms, timeAccumulator500ms);
            timeAccumulator500ms = 0;
        }
    }
}
```

**预期收益：**
- CPU使用减少70%
- 更平滑的帧率
